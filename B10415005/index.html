Race Condition

資源在同時間內被多個CPU存取，若至少有一個寫入動作，則可能導致其他CPU存取到的結果錯誤，此為Race Condition。
以下程式碼是Linked List簡單的實作範例。
insert函式新增元素到List最前方：


















 1 . struct list {
 2 . 　　int data;
 3 . 　　truct list *next;
 4 . };
 5 .
 6 . struct list *list = 0;
 7 .
 8 . void
 9 . insert(int data)
 10. {
 11. 　　struct list *l;
 12.
 13. 　　l = malloc(sizeof *l);
 14. 　　l->data = data;
 15. 　　l->next = list;
 16. 　　list = l;
 17. }

當兩個CPU同時進行insert時，在15、16行可能出現錯誤，兩個CPU都會同時將l->next指向當前的list，並且在第16行後，後執行的CPU會蓋掉他人的結果。

Lock&Critical Section

Lock可避免資源在多CPU存取下，因Race Condition而導致結果錯誤。
被Lock起來的區間稱為Critical Section，保證同時間只有一個CPU會執行Critical Section中的程式碼，並且保持該區域的不變性(Invariant)，利用這點可以對資源進行保護，將之前的程式碼些許改動，可以避免Race Condition產生的錯誤結果：
















 1 . struct list *list = 0;
 2 . struct lock listlock;
 3 .
 4 . void
 5 . insert(int data)
 6 . {
 7 . 　　struct list *l;
 8 .    acquire(&listlock);
 9 . 
 10. 　 l = malloc(sizeof l);
 11. 　 l->data = data;
 12. 　 l->next = list;
 13. 　 list = l;
 14. 　 release(&listlock);
 15. }

可以將Critical Section想像成序列化(Serialized)的程式碼區塊或者想像該區域間的指令都是原子(Atomic)操作指令，較晚獲得lock的thread或CPU，看到的都是前者完整執行完的結果。

Spin Lock

Overview

1. Data Structure Of Spin Lock

Xv6提供兩種lock類型，首先介紹spin-lock，以下是spin-lock的資料結構：









1. struct spinlock {
2.   uint locked; // 0代表lock可用，非0代表已被使用(held)
3. 
4.   // 以下用於debugging:
5.   char *name; // 代表此lock的名稱
6.   struct cpu *cpu; // 使用此lock的CPU
7.   uint pcs[10]; // 函式呼叫的stack，以PC(Program Counter)代表
8. };

2. Acquire

以下是acquire第一版的程式碼以及流程圖：











1 . void
2 . acquire(struct spinlock *lk)
3 . {
4 .   for(;;) {
5 .     if(!lk->locked) {
6 .       lk->locked = 1;
7 .       break;
8 .     }
9 .   }
10. }


但是這個版本的acquire本身就存在Race Condition的問題，因為可能同時多個CPU執行第5行，在當下條件皆成立，進而導致兩個CPU都獲取lock，這本身就違反了acquire的意圖。

要讓這段程式碼能夠實現原子操作(Atomic)，必須使用x86下的特殊指令xchg，這個指令用意在於將記憶體中的某個字組(Word)與特定的值進行交換。
以下的程式碼是修正過的acquire，也是xv6真正使用的程式碼，其中xchg(&lk−>locked, 1) 當lk->locked本來就是1，則回傳值也是1，若lk->locked內是0，則與1交換，並且xchg回傳0。



















1 . void
2 . acquire(struct spinlock *lk)
3 . {
4 .   pushcli(); // 為了避免deadlock，關閉中斷
5 .   if(holding(lk))
6        panic("acquire");
7 .
8 .   // 使用xchg原子操作指令實現之前的功能
9 .   while(xchg(&lk−>locked, 1) != 0)
10.   ;
11.
12.   // 告訴系統以及編譯器，不要重新安排指令執行順序
13.   __sync_synchronize();
14.
15.   // 以下設定debugging資訊
16.   lk−>cpu = mycpu();
17.   getcallerpcs(&lk, lk−>pcs);
18. }

Interrupts&Ordering

須注意到的是，如果啟用interrupts，則kernel code可能停止在任何階段，並且執行interrups handler，若interrup handler嘗試存取先前process已經取得的lock，則可能產生deadlock，xv6的保守作法是：spin-lock保護的Critical Section，都會停用interrupts，對應即為上述程式碼第四行，以下是pushcli的程式碼以及說明：












1 . void
2 . pushcli(void)
3 . {
4 .   int eflags;
5 .
6 .   eflags = readeflags(); //取得當前flag
7 .   cli(); //關閉interrupts
8 .   if(mycpu()−>ncli == 0)
9 .     mycpu()−>intena = eflags & FL_IF; //改變CPU flag，使其關閉interrupts
10.   mycpu()−>ncli += 1; //堆疊累加
11. }

CPU或者編譯器(Compiler)有時為了要最佳化CPU性能，會讓需要比較多Cycle的指令先執行，如此可以和其他指令進行Instruction pipelining，會在不影響結果的情況下重新安排指令執行順序，但是對於以下的程式碼，重新排序指令順序可能導致災難：







1. acquire(&listlock);
2. l = malloc(sizeof *l);
3. l->data = data;
4. l->next = list;
5. list = l;
6. release(&listlock);

如果重新排序順序，使得執行完第6行後，第3~5行被其他的執行緒看見，則可能導致嚴重的錯誤結果。
為了要避免這樣的問題，可以告知系統以及編譯器不要進行指令重排，__sync_synchronize()即是負責此功能。

3. Release
要釋放lock，必須清除lock保存的debugging資訊，並且將locked欄位設定回0，跟acquire相同，變更locked的動作本身也必須是原子操作(Atomic)才能保證正確性，x86保證movl可以安全地更新4Bytes資料。


















1. void
2. release(struct spinlock *lk)
3. {
4.   if(!holding(lk))
5.   panic("release");
6.
7.   lk−>pcs[0] = 0;
8.   lk−>cpu = 0;
9.
10.   // 告知不要重新編排指令執行順序
11.   __sync_synchronize();
12.
13.   // 可看作是 lk−>locked = 0 的原子指令
16.   asm volatile("movl $0, %0" : "+m" (lk−>locked) : );
17.
18.   popcli();
19. }

在acquire的時候有討論到spin-lock在進入Crititcal Section時會關閉interrupt，顯然地，在離開始必須將其開啟，popcli函式即是，以下為其程式碼&說明：











1 . void
2 . popcli(void)
3 . {
4 .   if(readeflags()&FL_IF) //檢查是否禁用interrupts
5 .     panic("popcli − interruptible");
6 .   if(−−mycpu()−>ncli < 0)
7 .     panic("popcli");
8 .   if(mycpu()−>ncli == 0 && mycpu()−>intena)
9 .     sti();
10. }

Sleep Lock

Overview

1. DataStructure Of Sleep Lock
有時候可能會長期持有lock，例如檔案系統，然而等待期間若要讓其他Process能夠持續進行，則必須要可以在Context-Switches環境下運作的lock機制，Xv6為此提供了sleep-lock。









1. struct sleeplock {
2.   uint locked; // 是否被使用了(held)?
3.   struct spinlock lk; // 用來保護sleep lock
4.
5.   // debugging資訊:
6.   char *name; // lock的名稱
7.   int pid; // 使用此lock的process
8. };

2. Acquiresleep
為了要空出CPU資源，acquiresleep會呼叫sleep函式，並且釋放自己擁有的spin-lock，如此其他thread可在等待期間繼續執行。
以下分別是acquire(sleep)以及release(sleep)的程式碼：












1 . void
2 . acquiresleep(struct sleeplock *lk)
3 . {
4 .   acquire(&lk−>lk);
5 .   while (lk−>locked) {
6 .     sleep(lk, &lk−>lk);
7 .   }
8 .   lk−>locked = 1;
9 .   lk−>pid = myproc()−>pid;
10.   release(&lk−>lk);
11. }










1. void
2. releasesleep(struct sleeplock *lk)
3. {
4.   acquire(&lk−>lk);
5.   lk−>locked = 0;
6.   lk−>pid = 0;
7.   wakeup(lk);
8.   release(&lk−>lk);
9. }

Lock的極限

Lock看似能夠解決多執行緒下的許多問題，但是仍存在一些缺陷。
假定某個函式foo，內有資料需要被保護，但foo被同時被兩種函式呼叫：已經取得lock以及無使用lock的函式，這種情況通常有兩種處理方法，一是讓呼叫者持有lock，或者讓被呼叫者持有lock，以下的程式碼中，wakeup1函式要求呼叫者持有lock，而最開始的Linked List Insert函式則是自己持有Lock。











1. void
2. wakeup1(void *chan)
3. {
4.   struct proc *p;
5.
6.   for(p = ptable.proc; p < &ptable.proc[NPROC]; p++)
7.     if(p−>state == SLEEPING && p−>chan == chan)
8.       p−>state = RUNNABLE;
9. }

不過，最簡單的解決方法是使用支持遞迴的Recursive Locks。
